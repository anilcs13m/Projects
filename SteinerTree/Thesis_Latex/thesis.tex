\newcommand\segment[4]{

}

\newcommand\lblsegmenttwo[5]{

\path (#3) edge[|-|] node[at start, above] {#5} (#4) ;
}

\newcommand\lblsegmentthree[5]{

}

\newcommand\lblsegmentfour[5]{
